Slide with Auto-shapes

[261]:

# Create the bullet point & picture slide:
slide4_register = prs.slide_layouts[5]
slide4 = prs.slides.add_slide(slide4_register)

# Add the title of this slide
title4 = slide4.shapes.title
title4.text = "Title of the slide N°4"

# Add the first shape (shape n°1)

# Positioning of the shape n°1
left1 = top1 = width1 = height1 = Inches(2)

# kind of the shape n°1: ROUND_RECTANGLE
add_shape1 = slide4.shapes.add_shape(MSO_SHAPE.ROUND_1_RECTANGLE, left1, top1, width1, height1)

# Add the second shape (shape n°2)

# Positioning of the shape n°1
left2 = Inches(6)
top2 = Inches(2)
width2 = height2 = Inches(2)

# kind of the shape n°2: ARROW
arrow1 = slide4.shapes.add_shape(MSO_SHAPE.DOWN_ARROW, left2, top2, width2, height2)

# change the main colour of the arrow
fill_arrow1 = arrow1.fill
fill_arrow1.solid() # solid colour
fill_arrow1.fore_color.theme_color = MSO_THEME_COLOR.ACCENT_2 # change the color

# rotate the arrow to 90 degrees
arrow1.rotation = 90
